  const gradeData = reviewPreview.grades.find((g) => g.grade === grade);

};

function formatMemoryScore(value) {
  const safeValue = Math.max(0, Math.min(1, Number(value || 0)));
  return `${Math.round(safeValue * 100)}%`;
}
