## Task 2: 封装 `usePretext` hook

**文件**: `frontend/src/hooks/usePretext.ts`（新建）

**内容**:

```ts
import { prepare, layout } from '@chenglou/pretext'

export function usePretext() {
  /**
   * 一次性文本预处理（测量 + 分段）
   * 相同 text + font 的结果会被 Pretext 内部缓存
   * @param text 英文文本
   * @param font CSS font shorthand，如 '16px Inter'
   */
  function prepareText(text: string, font: string) {
    return prepare(text, font)
  }

  /**
   * 根据已知宽度和行高计算总高度
   * 这是热路径，多次调用无性能损耗（纯算术）
   * @param prepared prepareText() 返回的句柄
   * @param maxWidth 最大宽度（px）
   * @param lineHeight 行高（px），需与 CSS line-height 一致
   */
  function calculateLayout(prepared: ReturnType<typeof prepare>, maxWidth: number, lineHeight: number) {
    return layout(prepared, maxWidth, lineHeight)
  }

  return { prepareText, calculateLayout }
}
```

**验证**: `import { usePretext } from '@/hooks/usePretext'` 无报错；`prepare('hello world', '16px Inter')` 返回非空对象。

**完成**: `usePretext.ts` 存在，TypeScript 编译无错。

## Task 3: 封装 `useRichLayout` hook（含 CEFR 元数据合并）

**文件**: `frontend/src/hooks/useRichLayout.ts`（新建）

**内容**:

核心思路：`prepareWithSegments` 返回的分段粒度是自然段落/单词边界，结合 `vocabAnalyzer` 的词级 CEFR 等级后，每个 segment 可携带额外元数据。

```ts
import { prepareWithSegments, layoutWithLines } from '@chenglou/pretext'
import { VocabAnalyzer } from '@/utils/vocabAnalyzer'

export interface RichSegment {
  text: string          // 分段文本
  cefrLevel: string | null  // CEFR 等级（A1-C2/SUPER/null）
  word: string          // 标准化词形（用于查表）
}

export interface RichLine {
  text: string          // 行文本
  width: number        // 行宽（px）
  segments: RichSegment[]  // 该行包含的分段（带 CEFR）
}

const DEFAULT_FONT = '16px Inter'
const DEFAULT_LINE_HEIGHT = 24

let _analyzerInstance: VocabAnalyzer | null = null
async function getAnalyzer(): Promise<VocabAnalyzer> {
  if (!_analyzerInstance) {
    _analyzerInstance = new VocabAnalyzer()
    await _analyzerInstance.load()
  }
  return _analyzerInstance
}

/**
 * 使用 VocabAnalyzer 对文本中每个词进行 CEFR 标注
 * @param text 纯文本
 * @returns 带 CEFR 等级的词列表
 */
async function enrichWithCefr(text: string): Promise<RichSegment[]> {
  const analyzer = await getAnalyzer()
  // VocabAnalyzer.analyzeSentence 内部 _tokenize 会过滤 stopwords，
  // 但我们需要的 rich layout 中所有词都要有 CEFR 信息，
  // 所以直接用 surface form 逐词查表，不经过 tokenize/stopwords 过滤
  const words = text.split(/\s+/).filter(w => w.length > 0)
  return words.map(word => {
    const level = analyzer.lookupCefrLevelForSurfaceForm(word)
    return {
      text: word,
      cefrLevel: level,
      word: word.toLowerCase().replace(/[^a-z]/g, ''),
    }
  })
}

/**
 * useRichLayout — CEFR-aware Pretext 布局 hook
 *
 * 流程：text + font → prepareWithSegments → layoutWithLines → RichLine[]
 *
 * 返回值包含：
 * - lines: RichLine[] — 每行的文本、宽度和带 CEFR 标注的分段
 * - isReady: boolean — 是否已完成首次测量
 * - reload: (text: string, maxWidth: number) => void — 重新测量
 */
export function useRichLayout(initialText: string, maxWidth: number) {
  const [lines, setLines] = useState<RichLine[]>([])
  const [isReady, setIsReady] = useState(false)
  const font = DEFAULT_FONT
  const lineHeight = DEFAULT_LINE_HEIGHT

  const compute = useCallback(async (text: string, width: number) => {
    if (!text.trim()) {
      setLines([])
      setIsReady(false)
      return
    }
    const prepared = prepareWithSegments(text, font)
    const result = layoutWithLines(prepared, width, lineHeight)
    const segments = await enrichWithCefr(text)
    // 按 grapheme 范围切分 segments 到 lines
    const richLines: RichLine[] = result.lines.map(line => ({
      text: line.text,
      width: line.width,
      segments: extractSegmentsForLine(segments, line),
    }))
    setLines(richLines)
    setIsReady(true)
  }, [font, lineHeight])

  useEffect(() => {
    compute(initialText, maxWidth)
  }, [initialText, maxWidth, compute])

  return { lines, isReady, reload: compute }
}
```

**辅助函数**（文件内部）:

```ts
function extractSegmentsForLine(allSegments: RichSegment[], line: LayoutLine): RichSegment[] {
  // line.start / line.end 是 grapheme cursor
  // 我们按单词 index 近似切分（足够精确用于 CEFR 展示）
  // 真实场景下需要 grapheme index 到 word index 的映射
  // 这里先用简化版本：直接返回所有 segments，后续精确切分在 Task 4 优化
  return allSegments
}
```

**验证**: `useRichLayout` hook 可正常调用，返回 `lines: RichLine[]` 和 `isReady: boolean`。

**完成**: `useRichLayout.ts` 存在，编译无错。

## Task 4: localStorage 缓存策略实现

**文件**: `frontend/src/hooks/usePretextCache.ts`（新建）

**内容**:

缓存 key 设计：`pt:prepare:${hash(text)}:${font}` → `PreparedText`

```ts
const CACHE_PREFIX = 'pt:prepare:'

function getCacheKey(text: string, font: string): string {
  // 简单的 hash（文本通常不长，直接用 JSON.stringify 再 SHA）
  const hash = btoa(encodeURIComponent(text)).slice(0, 32)
  return `${CACHE_PREFIX}${hash}:${font}`
}

function safeJsonParse<T>(str: string): T | null {
  try { return JSON.parse(str) } catch { return null }
}

/**
 * 使用 localStorage 缓存 prepare 结果
 * 缓存 key = text hash + font，命中则跳过 Pretext 内部测量
 */
export function usePretextCache() {
  function get(text: string, font: string): string | null {
    try {
      return localStorage.getItem(getCacheKey(text, font))
    } catch { return null }
  }

  function set(text: string, font: string, preparedJson: string) {
    try {
      localStorage.setItem(getCacheKey(text, font), preparedJson)
    } catch {
      // localStorage 写满时静默丢弃（用户仍可正常使用）
    }
  }

  function clear() {
    try {
      const keys = Object.keys(localStorage).filter(k => k.startsWith(CACHE_PREFIX))
      keys.forEach(k => localStorage.removeItem(k))
    } catch { /* ignore */ }
  }

  return { get, set, clear }
}
```

**验证**: `usePretextCache` 可正常读取/写入 localStorage；`clear()` 后缓存被清空。

**完成**: `usePretextCache.ts` 存在。
