## 步骤 3: 验证 computeCefrClassName 边界逻辑

```python
LEVEL_ORDER = ["A1", "A2", "B1", "B2", "C1", "C2", "SUPER"]

def get_level_index(level):
    try:
        return LEVEL_ORDER.index(level)
    except ValueError:
        return 6

def compute_cefr_class_name(word_level, user_level):
    if word_level is None or word_level == '':
        return "cefr-mastered"
    if word_level == 'SUPER':
        return "cefr-above-i-plus-one"
    word_idx = get_level_index(word_level)
    user_idx = get_level_index(user_level)
    if word_idx <= user_idx:
        return "cefr-mastered"
    if word_idx == user_idx + 1:
        return "cefr-i-plus-one"
    return "cefr-above-i-plus-one"

# TEST-03: 关键边界
tests = [
    (None,      'B1', 'cefr-mastered'),         # 词不在表里 → 灰
    ('A1',      'B1', 'cefr-mastered'),         # A1 <= B1 → 灰
    ('B1',      'B1', 'cefr-i-plus-one'),        # B1 = B1+0 → 绿
    ('B2',      'B1', 'cefr-i-plus-one'),        # B2 = B1+1 → 绿
    ('C1',      'B1', 'cefr-above-i-plus-one'),  # C1 >= B1+2 → 红
    ('C2',      'B1', 'cefr-above-i-plus-one'),  # C2 >= B1+2 → 红
    ('SUPER',   'B1', 'cefr-above-i-plus-one'),  # SUPER → 始终红
    ('SUPER',   'C2', 'cefr-above-i-plus-one'),  # SUPER 即使 C2 用户 → 始终红
    ('A1',      'A1', 'cefr-mastered'),         # A1 = A1+0 → 绿（i+0）
    ('A2',      'A1', 'cefr-i-plus-one'),        # A2 = A1+1 → 绿
    ('',        'B1', 'cefr-mastered'),          # 空字符串 → 灰
]

all_passed = True
for word_level, user_level, expected in tests:
    result = compute_cefr_class_name(word_level, user_level)
    status = "OK" if result == expected else "FAIL"
    print(f"  [TEST-03] computeCefrClassName({word_level!r}, {user_level!r}) = {result!r} [{status}]")
    if result != expected:
        all_passed = False

assert all_passed, "Some tests failed"
print(f"  [TEST-03] computeCefrClassName boundary tests: ALL PASSED OK")
```

def simulate_analyze_sentence(words_map, sentence):
    tokens = [w.strip(".,!?'\"") for w in sentence.split() if len(w.strip(".,!?'\"")) > 1]
    level_counts = {lvl: 0 for lvl in ['A1','A2','B1','B2','C1','C2','SUPER']}
    total_rank = 0
    known_count = 0
    for token in tokens:
        lower = token.lower()
        if lower in words_map:
            info = words_map[lower]
            lvl = info.get('level', 'SUPER')
            rank = info.get('rank', 0)
            level_counts[lvl] = level_counts.get(lvl, 0) + 1
            total_rank += rank
            known_count += 1
        else:
            level_counts['SUPER'] += 1
    avg_rank = total_rank / known_count if known_count > 0 else 0
    return level_counts, avg_rank

test_sentences = [
    "The quick brown fox jumps over the lazy dog.",
    "I am going to the store to buy some groceries.",
    "Quantum computing is a fascinating field of study.",
    "She elegantly danced across the stage with remarkable grace.",
    "The meeting will commence at nine o clock in the morning.",
]

for sentence in test_sentences:
    lc, avg_rank = simulate_analyze_sentence(words, sentence)
    print(f"  [TEST-02] '{sentence[:50]}...'")
    print(f"    counts: {dict(sorted(lc.items()))}, avg_rank={avg_rank:.0f}")

print(f"  [TEST-02] analyzeSentence simulation: no crashes OK")
