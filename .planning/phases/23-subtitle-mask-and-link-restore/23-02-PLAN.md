       ```javascript
       async function submitLinkRestore(lesson) {
         if (!lesson?.source_url) return;
         
         setRestoringLessonId(lesson.id);
         setStatus("");
         try {
           // Trigger yt-dlp re-download via desktop bridge
           const response = await requestDesktopLocalHelper(
             "/api/desktop-asr/url-import/tasks",
             "json",
             {
               method: "POST",
               headers: { "Content-Type": "application/json" },
               body: { source_url: lesson.source_url },
             }
           );
           
           if (response?.ok) {
             setStatus("视频已更新");
             // Refresh the lesson media
             if (onRestoreMedia) {
               // The desktop bridge should have already updated the IndexedDB cache
               // Trigger a refresh by calling onRestoreMedia with null to signal refresh
               await onRestoreMedia(lesson, null);
             }
           } else {
             setStatus(response?.message || "按链接恢复失败");
           }
         } catch (error) {
           setStatus(error?.message || "按链接恢复失败");
         } finally {
           setRestoringLessonId(null);
           restoreTargetRef.current = null;
         }
       }
       ```
