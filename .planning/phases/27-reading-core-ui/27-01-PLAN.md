### Plan 02 — ArticlePanel 主体渲染

**目标**：将 `RichLine[]` 渲染为 flex 行流，实现方案 A 文章主体区。

**新建文件**：`frontend/src/features/reading/ArticlePanel.jsx`

**依赖**：Plan 01（目录结构已建立）

**具体操作**：

1. 实现 `ArticlePanel` 组件：

```jsx
import { useEffect, useRef, useCallback, useState } from "react";
import { useRichLayout } from "@/hooks/useRichLayout";
import { readCefrLevel } from "@/app/authStorage";
import { cn } from "@/lib/utils";
import "./reading.css";

const ARTICLE_FONT = "18px Inter";
const ARTICLE_LINE_HEIGHT = 30;

export function ArticlePanel({ text, contentWidth, onWidthChange }) {
  const containerRef = useRef(null);
  const [measuredWidth, setMeasuredWidth] = useState(contentWidth);
  const userLevel = readCefrLevel() || "B1";

  // 监听容器宽度变化
  useEffect(() => {
    if (!containerRef.current) return;
    const observer = new ResizeObserver((entries) => {
      for (const entry of entries) {
        const w = entry.contentRect.width;
        setMeasuredWidth(w);
        onWidthChange?.(w);
      }
    });
    observer.observe(containerRef.current);
    return () => observer.disconnect();
  }, [onWidthChange]);

  const { lines, isReady, error } = useRichLayout(text, measuredWidth, ARTICLE_FONT, ARTICLE_LINE_HEIGHT);

  if (!isReady) {
    return <div className="article-panel-loading">正在加载文章...</div>;
  }

  if (error) {
    return <div className="article-panel-error">加载失败: {error}</div>;
  }

  return (
    <div ref={containerRef} className="article-panel">
      <div className="article-content">
        {lines.map((line, lineIdx) => (
          <div key={lineIdx} className="article-line">
            {line.segments.map((seg, segIdx) => (
              <ArticleWord key={segIdx} segment={seg} userLevel={userLevel} />
            ))}
          </div>
        ))}
      </div>
    </div>
  );
}
```

2. 在 `ArticlePanel.jsx` 底部添加 `ArticleWord` 子组件（词级渲染，后续 Plan 03 扩展）：

```jsx
function ArticleWord({ segment, userLevel }) {
  const cefrLevel = segment.cefrLevel;
  const cefrClass = computeCefrClassName(cefrLevel, userLevel);
  return (
    <span className={cn("article-word", cefrClass)} title={cefrLevel || "未知"}>
      {segment.text}
    </span>
  );
}
```

需要从 `CefrBadge.jsx` 导入 `computeCefrClassName`。

**验收条件**：

- ResizeObserver 监听容器宽度，`measuredWidth` 变化时触发 `useRichLayout.reload()`
- 每个词渲染为 `<span class="article-word">` 带 CEFR class

3. 更新 `ReadingPage.jsx` 整合 WordSidebar：

```jsx
export function ReadingPage({ accessToken }) {
  const [contentWidth, setContentWidth] = useState(600);
  const [selectedWords, setSelectedWords] = useState([]);

  function handleWordClick(word, segment) {
    const already = selectedWords.find((w) => w.word === word);
    if (already) {
      setSelectedWords((prev) => prev.filter((w) => w.word !== word));
    } else {
      setSelectedWords((prev) => [
        ...prev,
        { word, cefrLevel: segment.cefrLevel, cefrClass: segment.cefrClass },
      ]);
    }
  }

  return (
    <div className="reading-layout">
      <ArticlePanel text={DEMO_TEXT} contentWidth={contentWidth} onWidthChange={setContentWidth} onWordClick={handleWordClick} />
      <WordSidebar selectedWords={selectedWords} onRemove={(item) => setSelectedWords((prev) => prev.filter((w) => w.word !== item.word))} />
    </div>
  );
}
```

4. 更新 `ArticleWord` 组件以支持 `selected` 态：
