## 步骤 5: 验证输出

运行验证脚本（创建并执行）：
```python
import json

with open('app/data/vocab/cefr_vocab_fixed.json', encoding='utf-8') as f:
    data = json.load(f)

# 1. 检查版本字段
assert '_vocab_version' in data, "Missing _vocab_version"
assert data['_vocab_version'] == 'fixed-v1', f"Wrong version: {data['_vocab_version']}"

# 2. 检查词数
words = data.get('words', {})
assert len(words) == 50000, f"Word count mismatch: {len(words)} != 50000"

# 3. 检查无 null 值
null_words = [w for w, info in words.items() if info.get('level') is None]
assert len(null_words) == 0, f"Null level words: {null_words[:5]}"

# 4. 检查 CEFR-J 匹配词有 pos_entries
matched_with_pos = [w for w, info in words.items() if info.get('pos_entries')]
matched_with_source = [w for w, info in words.items() if info.get('_source') == 'CEFR-J']
assert len(matched_with_pos) >= 6000, f"Too few words with pos_entries: {len(matched_with_pos)}"
assert len(matched_with_source) >= 6000, f"Too few CEFR-J matched words: {len(matched_with_source)}"

# 5. 检查未匹配词有 _source 标记
unmatched = [w for w, info in words.items() if info.get('_source') == 'rank-based']
assert len(unmatched) >= 40000, f"Too few rank-based words: {len(unmatched)}"

# 6. 检查等级分布合理
level_dist = {}
for info in words.values():
    lvl = info.get('level', 'UNKNOWN')
    level_dist[lvl] = level_dist.get(lvl, 0) + 1
print("Level distribution:")
for lvl, cnt in sorted(level_dist.items()):
    print(f"  {lvl}: {cnt}")

print("\nAll validations PASSED")
```
