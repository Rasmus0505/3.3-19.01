1. 在 `llm.py` 末尾添加新 endpoint：

```python
from app.infra.llm.deepseek import call_deepseek

REWRITE_SYSTEM_PROMPT = (
    "You are an English text simplifier for language learners.\n"
    "Rewrite the given text at CEFR {target_level} level.\n"
    "Rules:\n"
    "- Replace complex vocabulary with simpler CEFR {target_level} equivalents\n"
    "- Keep sentence structure clear and understandable\n"
    "- Preserve the original meaning and key information\n"
    "- Output only the rewritten text, no explanations or markers\n"
    "- Keep approximately the same length as the original\n"
)

REWRITE_MAX_INPUT_TOKENS = 3000  # 约等于 2000 英文词原文
REWRITE_MAX_OUTPUT_TOKENS = 2048


@router.post(
    "/rewrite-text",
    responses={503: {"model": ErrorResponse}, 402: {"model": ErrorResponse}, 413: {"model": ErrorResponse}},
)
def rewrite_text_endpoint(
    text: str = Body(..., max_length=REWRITE_MAX_INPUT_TOKENS * 4),  # ~12KB
    target_level: str = Query(default="B1", max_length=4),
    enable_thinking: bool = Query(default=False),
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db),
):
    """
    Rewrite given English text at CEFR target_level using DeepSeek V3.2.
    Charges the user according to the selected model rate.
    """
    if target_level.upper() not in CEFR_LEVELS:
        raise HTTPException(
            status_code=422,
            detail=f"Invalid target_level '{target_level}'. Must be one of: {', '.join(sorted(CEFR_LEVELS))}",
        )

    if not text or not text.strip():
        raise HTTPException(status_code=422, detail="text must be a non-empty string")

    effective_model = LLM_MODEL_DEEPSEEK_THINKING if enable_thinking else LLM_MODEL_DEEPSEEK_FAST

    try:
        rate = get_model_rate(db, effective_model)
    except Exception:
        raise HTTPException(status_code=503, detail="LLM model not available")

    api_key = _require_api_key()
    trace_id = str(uuid.uuid4())

    system_prompt = REWRITE_SYSTEM_PROMPT.format(target_level=target_level.upper())
    messages = [
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": text.strip()},
    ]

    try:
        rewritten_text, usage = call_deepseek(
            messages=messages,
            api_key=api_key,
            enable_thinking=enable_thinking,
            stream=False,
            temperature=0.3,  # 降低随机性，保持一致性
            max_tokens=REWRITE_MAX_OUTPUT_TOKENS,
        )
    except Exception as exc:
        logger.exception("[DEBUG] llm.rewrite_failed user_id=%s error=%s", current_user.id, str(exc)[:200])
        raise HTTPException(status_code=502, detail=f"LLM call failed: {str(exc)[:200]}")

    if not rewritten_text:
        raise HTTPException(status_code=502, detail="LLM returned empty result")

    total_tokens = usage.prompt_tokens + usage.completion_tokens

    from app.services.billing_service import calculate_llm_charge_by_tokens

    charge_cents = calculate_llm_charge_by_tokens(
        total_tokens=total_tokens,
        points_per_1k_tokens=rate.points_per_1k_tokens,
    )

    try:
        consume_points(
            db,
            user_id=current_user.id,
            points=charge_cents,
            model_name=effective_model,
            lesson_id=None,
            event_type=EVENT_CONSUME_LLM,
            note=f"重写文本，total_tokens={total_tokens}, enable_thinking={enable_thinking}",
        )
    except Exception:
        pass  # 计费失败不影响重写结果返回

    from app.services.llm_usage_service import log_llm_usage

    log_llm_usage(
        db,
        user_id=current_user.id,
        model_name=effective_model,
        category="rewrite",
        prompt_tokens=usage.prompt_tokens,
        completion_tokens=usage.completion_tokens,
        reasoning_tokens=usage.reasoning_tokens,
        total_tokens=total_tokens,
        input_cost_cents=None,
        charge_cents=charge_cents,
        lesson_id=None,
        enable_thinking=enable_thinking,
        input_text_preview=text.strip()[:200],
        trace_id=trace_id,
    )

    db.commit()

    return {
        "ok": True,
        "rewritten_text": rewritten_text,
        "model": effective_model,
        "usage": {
            "prompt_tokens": usage.prompt_tokens,
            "completion_tokens": usage.completion_tokens,
            "reasoning_tokens": usage.reasoning_tokens,
            "total_tokens": total_tokens,
        },
        "charge_cents": charge_cents,
        "trace_id": trace_id,
    }
```

### Plan 02 — 前端 IndexedDB 存储 + API 调用 hook

**目标**：创建 `useReadingRewrite` hook，实现重写 API 调用和 IndexedDB 本地存储。

**新建文件**：`frontend/src/hooks/useReadingRewrite.js`

**具体操作**：

1. 创建 IndexedDB wrapper（参考 `localTaskStore.js`）：

```js
const DB_NAME = "reading_rewrites";
const DB_VERSION = 1;
const STORE_NAME = "rewrites";

function openRewriteDB() {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, DB_VERSION);
    req.onerror = () => reject(req.error);
    req.onsuccess = () => resolve(req.result);
    req.onupgradeneeded = (e) => {
      const db = e.target.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        const store = db.createObjectStore(STORE_NAME, { keyPath: "id" });
        store.createIndex("lesson_id", "lesson_id", { unique: false });
        store.createIndex("created_at", "created_at", { unique: false });
      }
    };
  });
}

export async function saveRewrite(record) {
  const db = await openRewriteDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE_NAME, "readwrite");
    const store = tx.objectStore(STORE_NAME);
    const req = store.put(record);
    req.onsuccess = () => resolve(record.id);
    req.onerror = () => reject(req.error);
  });
}

export async function getRewriteById(id) {
  const db = await openRewriteDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE_NAME, "readonly");
    const store = tx.objectStore(STORE_NAME);
    const req = store.get(id);
    req.onsuccess = () => resolve(req.result || null);
    req.onerror = () => reject(req.error);
  });
}

export async function getLatestRewrite() {
  const db = await openRewriteDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE_NAME, "readonly");
    const store = tx.objectStore(STORE_NAME);
    const index = store.index("created_at");
    const req = index.openCursor(null, "prev");
    req.onsuccess = () => {
      const cursor = req.result;
      resolve(cursor ? cursor.value : null);
    };
    req.onerror = () => reject(req.error);
  });
}

export async function clearAllRewrites() {
  const db = await openRewriteDB();
  return new Promise((resolve, reject) => {
    const tx = db.transaction(STORE_NAME, "readwrite");
    const store = tx.objectStore(STORE_NAME);
    const req = store.clear();
    req.onsuccess = () => resolve();
    req.onerror = () => reject(req.error);
  });
}
```

2. 创建 `useReadingRewrite` hook：

```js
import { useCallback, useState } from "react";
import { parseResponse } from "../../shared/api/client";
import { saveRewrite, getLatestRewrite } from "./useReadingRewrite";
import { readCefrLevel } from "../../app/authStorage";

/**
 * useReadingRewrite — 阅读板块重写状态管理
 *
 * @param {object} props
 * @param {Function} props.apiCall — API 调用函数
 * @param {string} props.accessToken — 用户 access token
 */
export function useReadingRewrite({ apiCall, accessToken }) {
  const [rewrittenText, setRewrittenText] = useState(null);
  const [rewriteId, setRewriteId] = useState(null);
  const [viewMode, setViewMode] = useState("original"); // "original" | "rewritten"
  const [isRewriting, setIsRewriting] = useState(false);
  const [rewriteError, setRewriteError] = useState(null);

  const handleRewrite = useCallback(async (originalText) => {
    if (!accessToken) {
      const { toast } = await import("sonner");
      toast.error("请先登录");
      return;
    }
    if (!apiCall) {
      const { toast } = await import("sonner");
      toast.error("无法发起请求");
      return;
    }

    setIsRewriting(true);
    setRewriteError(null);

    try {
      const userLevel = readCefrLevel() || "B1";
      const cefrOrder = ["A1", "A2", "B1", "B2", "C1", "C2"];
      const userIdx = cefrOrder.indexOf(userLevel);
      const targetIdx = Math.min(userIdx + 1, cefrOrder.length - 1);
      const targetLevel = cefrOrder[targetIdx];

      const resp = await apiCall("/api/llm/rewrite-text", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          text: originalText,
          target_level: targetLevel,
          enable_thinking: false,
        }),
      });

      const data = await parseResponse(resp);

      if (!resp.ok || !data.ok || !data.rewritten_text) {
        const { toast } = await import("sonner");
        toast.error("重写失败：" + (data?.message || "未知错误"));
        setRewriteError(data?.message || "重写失败");
        return;
      }

      // 存储到 IndexedDB
      const id = await saveRewrite({
        id: data.trace_id || crypto.randomUUID(),
        lesson_id: null,
        original_text: originalText,
        rewritten_text: data.rewritten_text,
        target_level: targetLevel,
        user_level: userLevel,
        created_at: Date.now(),
      });

      setRewriteId(id);
      setRewrittenText(data.rewritten_text);
      setViewMode("rewritten");

      const { toast } = await import("sonner");
      toast.success("重写完成，消耗 " + (data.charge_cents / 100).toFixed(2) + " 元");
    } catch (err) {
      const { toast } = await import("sonner");
      toast.error("重写失败：" + (err?.message || "网络错误"));
      setRewriteError(err?.message || "网络错误");
    } finally {
      setIsRewriting(false);
    }
  }, [accessToken, apiCall]);

  const handleSwitchView = useCallback((mode) => {
    if (mode === "rewritten" && !rewrittenText) return;
    setViewMode(mode);
  }, [rewrittenText]);

  return {
    rewrittenText,
    rewriteId,
    viewMode,
    setViewMode: handleSwitchView,
    isRewriting,
    rewriteError,
    handleRewrite,
  };
}
```

**验收条件**：
- IndexedDB `reading_rewrites` DB 在首次调用时自动创建
- 重写完成后数据存储到 IndexedDB
- `handleRewrite` 返回 Promise，调用方可以 await
