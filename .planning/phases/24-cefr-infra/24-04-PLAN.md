    6. Add a useEffect for CEFR analysis triggered on lesson load. Find the existing useEffect that handles LESSON_LOADED phase. In that effect (or add a new one), add CEFR analysis:

       ```javascript
       useEffect(() => {
         if (phase !== "answering" || !sentences || sentences.length === 0) return;
         const lessonId = lesson?.id;
         if (!lessonId) return;

         const cacheKey = CEFR_CACHE_KEY_PREFIX + String(lessonId);
         const cached = localStorage.getItem(cacheKey);
         if (cached) {
           // Already analyzed — just mark as complete
           setCefrAnalysisStatus("complete");
           return;
         }

         // Need to analyze
         setCefrAnalysisStatus("analyzing");

         async function runAnalysis() {
           const analyzer = cefrAnalyzerRef.current;
           if (!analyzer) return;

           // Ensure vocab is loaded (happens once per session)
           if (!analyzer.isLoaded) {
             await analyzer.load();
           }

           const userLevel = useAppStore.getState().cefrLevel || "B1";
           const allSentences = sentences.map((s) => s.text_en).filter(Boolean);

           // Chunked processing to avoid UI stutter
           const results = [];
           for (let i = 0; i < allSentences.length; i += CEFR_ANALYSIS_CHUNK_SIZE) {
             const chunk = allSentences.slice(i, i + CEFR_ANALYSIS_CHUNK_SIZE);
             // Process chunk synchronously
             for (const sentence of chunk) {
               results.push(analyzer.analyzeSentence(sentence));
             }
             // Yield to UI thread between chunks
             if (i + CEFR_ANALYSIS_CHUNK_SIZE < allSentences.length) {
               await new Promise((resolve) => setTimeout(resolve, 0));
             }
           }

           // Build video report from chunked results
           const videoReport = analyzer.analyzeVideo(allSentences, userLevel);

           // Cache the results
           try {
             localStorage.setItem(cacheKey, JSON.stringify(videoReport));
           } catch (e) {
             // localStorage quota exceeded — ignore, analysis still works
             console.warn("[CEFR] Failed to cache analysis:", e);
           }

           setCefrAnalysisStatus("complete");

           // Show toast on first analysis completion
           toast.success("词汇分析完成", { duration: 2000 });
         }

         void runAnalysis();
       }, [phase, sentences, lesson?.id]);
       ```

    Per D-02: setTimeout(0) chunking prevents UI blocking.
    Per D-03: Toast "词汇分析完成" shown once after first analysis.
    Per D-08: Triggered on mount (phase === "answering" with sentences).
