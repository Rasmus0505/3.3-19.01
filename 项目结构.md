Dockerfile 配置
这是一个多阶段构建的应用：

dockerfile


# 第一阶段：构建前端
FROM node:22-alpine AS frontend-builder
WORKDIR /frontend
COPY frontend/package*.json ./
RUN npm ci
COPY frontend/ ./
RUN npm run build
# 第二阶段：Python 后端运行环境
FROM python:3.11-slim
LABEL "language"="python"
LABEL "framework"="fastapi"
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1
ENV PORT=8080
WORKDIR /app
# 安装系统依赖（包括 FFmpeg）
RUN apt-get update \
    && apt-get install -y --no-install-recommends ffmpeg ca-certificates \
    && rm -rf /var/lib/apt/lists/*
# 安装 Python 依赖
COPY requirements.txt ./
RUN pip install --no-cache-dir -r requirements.txt
# 复制应用文件
COPY alembic.ini ./
COPY migrations ./migrations
COPY app ./app
COPY --from=frontend-builder /frontend/dist/ ./app/static/
EXPOSE 8080
# 启动命令：支持自动数据库迁移
CMD ["sh", "-c", "if [ \"${AUTO_MIGRATE_ON_START:-1}\" != \"0\" ]; then python -m app.db.migration_bootstrap; fi && exec uvicorn app.main:app --host 0.0.0.0 --port 8080"]

✅ 前端 (Node.js) 和后端 (Python) 分离构建
✅ 包含 FFmpeg 支持（用于音视频处理）
✅ 支持自动数据库迁移（通过 AUTO_MIGRATE_ON_START 环境变量控制）
✅ 使用 Alembic 进行数据库版本管理
